    <script>
        // Inicializar contexto de audio de manera perezosa tras el primer click del usuario para evitar bloqueos del navegador
        let audioCtx = null;

        function initAudio() {
            if (!audioCtx) {
                audioCtx = new (window.AudioContext || window.webkitAudioContext)();
            }
        }

        // Sintetizador de tonos retro/arcade de 8 bits
        function playRetroSound(type) {
            initAudio();
            if (!audioCtx) return;

            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.connect(gain);
            gain.connect(audioCtx.destination);

            const now = audioCtx.currentTime;

            if (type === 'click') {
                // Sonido corto de UI (Minecraft-like click)
                osc.type = 'triangle';
                osc.frequency.setValueAtTime(150, now);
                osc.frequency.exponentialRampToValueAtTime(80, now + 0.1);
                gain.gain.setValueAtTime(0.3, now);
                gain.gain.exponentialRampToValueAtTime(0.01, now + 0.1);
                osc.start(now);
                osc.stop(now + 0.1);
            } else if (type === 'correct') {
                // Ding de nivel brillante de Minecraft
                osc.type = 'sine';
                osc.frequency.setValueAtTime(523.25, now); // Nota C5
                osc.frequency.setValueAtTime(659.25, now + 0.08); // Nota E5
                osc.frequency.setValueAtTime(783.99, now + 0.16); // Nota G5
                osc.frequency.setValueAtTime(1046.50, now + 0.24); // Nota C6
                
                gain.gain.setValueAtTime(0.15, now);
                gain.gain.exponentialRampToValueAtTime(0.01, now + 0.5);
                osc.start(now);
                osc.stop(now + 0.5);
            } else if (type === 'incorrect') {
                // Sonido de golpe "Oof" o daño de Minecraft
                osc.type = 'sawtooth';
                osc.frequency.setValueAtTime(120, now);
                osc.frequency.linearRampToValueAtTime(60, now + 0.25);
                
                gain.gain.setValueAtTime(0.2, now);
                gain.gain.exponentialRampToValueAtTime(0.01, now + 0.3);
                osc.start(now);
                osc.stop(now + 0.3);
            } else if (type === 'victory') {
                // Fanfarria triunfal corta
                osc.type = 'square';
                const notes = [261.63, 329.63, 392.00, 523.25, 392.00, 523.25];
                const durations = [0.1, 0.1, 0.1, 0.15, 0.1, 0.4];
                let currentStart = now;

                notes.forEach((freq, idx) => {
                    const localOsc = audioCtx.createOscillator();
                    const localGain = audioCtx.createGain();
                    localOsc.type = 'triangle';
                    localOsc.frequency.value = freq;
                    localOsc.connect(localGain);
                    localGain.connect(audioCtx.destination);
                    
                    localGain.gain.setValueAtTime(0.15, currentStart);
                    localGain.gain.exponentialRampToValueAtTime(0.01, currentStart + durations[idx]);
                    
                    localOsc.start(currentStart);
                    localOsc.stop(currentStart + durations[idx]);
                    currentStart += durations[idx] - 0.02;
                });
            } else if (type === 'gameover') {
                // Melodia triste descendente
                osc.type = 'sawtooth';
                osc.frequency.setValueAtTime(180, now);
                osc.frequency.linearRampToValueAtTime(120, now + 0.2);
                osc.frequency.linearRampToValueAtTime(90, now + 0.5);

                gain.gain.setValueAtTime(0.2, now);
                gain.gain.exponentialRampToValueAtTime(0.01, now + 0.6);
                osc.start(now);
                osc.stop(now + 0.6);
            }
        }
    </script>

        // Generador inteligente de preguntas sobre conceptos de potenciación
        function generateConceptQuestion() {
            // Tipos de pregunta:
            // 0: Identificar base
            // 1: Identificar exponente
            // 2: Identificar potencia (el resultado)
            // 3: Calcular la potencia dados base y exponente
            // 4: Calcular exponente dada base y potencia
            // 5: Calcular base dada potencia y exponente
            const type = Math.floor(Math.random() * 6);
            
            // Bases y exponentes pequeños para cálculo mental rápido, retador pero justo
            const pairs = [
                {b: 2, e: 3, p: 8},
                {b: 2, e: 4, p: 16},
                {b: 2, e: 5, p: 32},
                {b: 3, e: 2, p: 9},
                {b: 3, e: 3, p: 27},
                {b: 4, e: 2, p: 16},
                {b: 4, e: 3, p: 64},
                {b: 5, e: 2, p: 25},
                {b: 5, e: 3, p: 125},
                {b: 6, e: 2, p: 36},
                {b: 7, e: 2, p: 49},
                {b: 8, e: 2, p: 64},
                {b: 9, e: 2, p: 81},
                {b: 10, e: 2, p: 100},
                {b: 10, e: 3, p: 1000}
            ];

            const selectedPair = pairs[Math.floor(Math.random() * pairs.length)];
            const {b, e, p} = selectedPair;

            let questionText = "";
            let mathString = "";
            let correctAnswer = "";
            let options = [];

            switch(type) {
                case 0: // Identificar base
                    questionText = "En la siguiente expresión de potenciación, ¿cuál de los componentes es la <span class='text-green-400 font-bold'>BASE</span>?";
                    mathString = `${b}^{${e}} = ${p}`;
                    correctAnswer = `${b}`;
                    options = [correctAnswer, `${e}`, `${p}`, `${b * e}`];
                    break;
                case 1: // Identificar exponente
                    questionText = "En la siguiente ecuación cósmica, ¿cuál de los elementos corresponde al <span class='text-blue-400 font-bold'>EXPONENTE</span>?";
                    mathString = `${b}^{${e}} = ${p}`;
                    correctAnswer = `${e}`;
                    options = [correctAnswer, `${b}`, `${p}`, `${b + e}`];
                    break;
                case 2: // Identificar potencia
                    questionText = "En esta operación espacial, ¿cuál de los términos numéricos representa la <span class='text-purple-400 font-bold'>POTENCIA</span>?";
                    mathString = `${b}^{${e}} = ${p}`;
                    correctAnswer = `${p}`;
                    options = [correctAnswer, `${b}`, `${e}`, `${b * e}`];
                    break;
                case 3: // Calcular potencia completa
                    questionText = "Tienes un generador de energía de <span class='text-green-400'>Base</span> " + b + " y un acelerador de <span class='text-blue-400'>Exponente</span> " + e + ". ¿Cuánta <span class='text-purple-400'>Potencia</span> total produce?";
                    mathString = `${b}^{${e}} = x`;
                    correctAnswer = `${p}`;
                    options = [correctAnswer, `${b * e}`, `${p + 4}`, `${p - b}`];
                    break;
                case 4: // Hallar exponente
                    questionText = "Si el reactor espacial tiene una <span class='text-green-400'>Base</span> de " + b + " y produce una <span class='text-purple-400'>Potencia</span> de " + p + ", ¿cuál debe ser el <span class='text-blue-400'>Exponente</span>?";
                    mathString = `${b}^{x} = ${p}`;
                    correctAnswer = `${e}`;
                    options = [correctAnswer, `${e + 1}`, `${Math.round(p/b)}`, `${e - 1}`];
                    break;
                case 5: // Hallar base
                    questionText = "Un núcleo solar tiene un <span class='text-blue-400'>Exponente</span> de " + e + " y produce una <span class='text-purple-400'>Potencia</span> total de " + p + ". ¿Cuál es su <span class='text-green-400'>Base</span>?";
                    mathString = `x^{${e}} = ${p}`;
                    correctAnswer = `${b}`;
                    options = [correctAnswer, `${b + 2}`, `${Math.round(p/e)}`, `${b - 1}`];
                    break;
            }

            // Sanitizar opciones duplicadas o vacías que puedan ocurrir debido a fórmulas matemáticas
            options = [...new Set(options)].filter(opt => opt !== "");
            // Si por casualidad quedamos con menos de 4 opciones, inyectar aleatorios
            while (options.length < 4) {
                const randomVal = Math.floor(Math.random() * 20) + 1;
                if (!options.includes(`${randomVal}`)) {
                    options.push(`${randomVal}`);
                }
            }

            // Mezclar opciones
            options = options.sort(() => Math.random() - 0.5);

            return {
                questionText,
                mathString,
                correctAnswer,
                options
            };
        }

        function nextChallenge() {
            // Comprobar si ya ganó
            if (score >= 10) {
                endGame(true);
                return;
            }

            // Limpiar feedback anterior
            const fb = document.getElementById("feedback-alert");
            fb.classList.add("hidden");

            activeQuestion = generateConceptQuestion();

            // Renderizar pregunta e inicializar KaTeX en el contenedor correspondiente
            document.getElementById("question-text").innerHTML = activeQuestion.questionText;
            
            // Renderizar la fórmula principal con KaTeX de forma dinámica y controlada
            if (window.katex) {
                katex.render(activeQuestion.mathString, document.getElementById("math-display"), {
                    throwOnError: false,
                    displayMode: true
                });
            } else {
                document.getElementById("math-display").textContent = activeQuestion.mathString;
            }

            // Renderizar las opciones como botones de Minecraft
            const container = document.getElementById("options-container");
            container.innerHTML = "";
            activeQuestion.options.forEach(option => {
                const btn = document.createElement("button");
                btn.className = "w-full py-4 text-sm font-pixel text-white mc-button-gray tracking-wide flex items-center justify-center gap-2 transition duration-150 transform active:scale-95";
                btn.innerHTML = `<span class="bg-[#111] text-yellow-400 px-2.5 py-1 text-[10px] rounded border border-gray-600 block shadow-inner">${option}</span> Seleccionar`;
                btn.onclick = () => selectOption(option);
                container.appendChild(btn);
            });
        }

        function selectOption(selectedVal) {
            totalAttempts++;
            const fb = document.getElementById("feedback-alert");
            fb.classList.remove("hidden", "bg-green-900/80", "border-green-500", "text-green-300", "bg-red-900/80", "border-red-500", "text-red-300");

            if (selectedVal === activeQuestion.correctAnswer) {
                // Correcto
                playRetroSound('correct');
                score++;
                fb.classList.add("bg-green-900/80", "border-green-500", "text-green-300");
                fb.innerHTML = `¡SÍ! Núcleo procesado correctamente. ¡Energía Redstone en aumento!`;
                
                // Deshabilitar botones para evitar multi-clicks de trampas
                const btns = document.getElementById("options-container").getElementsByTagName("button");
                for (let button of btns) {
                    button.disabled = true;
                    button.style.opacity = "0.5";
                }

                updateHUD();
                setTimeout(nextChallenge, 1500);
            } else {
                // Incorrecto
                playRetroSound('incorrect');
                lives--;
                fb.classList.add("bg-red-900/80", "border-red-500", "text-red-300");
                fb.innerHTML = `¡BOOM! El reactor se sobrecalentó. La respuesta correcta era: <strong>${activeQuestion.correctAnswer}</strong>.`;
                
                const btns = document.getElementById("options-container").getElementsByTagName("button");
                for (let button of btns) {
                    button.disabled = true;
                    button.style.opacity = "0.5";
                }

                updateHUD();
                
                if (lives <= 0) {
                    setTimeout(() => endGame(false), 1500);
                } else {
                    setTimeout(nextChallenge, 2200);
                }
            }
        }

        function showInstructions() {
            playRetroSound('click');
            document.getElementById("screen-results").classList.add("hidden");
            document.getElementById("screen-instructions").classList.remove("hidden");
            
            // Re-renderizar expresiones LaTeX dentro del manual
            setTimeout(() => {
                if (window.katex) {
                    // Renderizar automáticamente elementos en el manual
                    const elements = document.getElementById("screen-instructions").querySelectorAll("p, li, span");
                    elements.forEach(el => {
                        // Buscar patrones en línea para renderizarlos de forma segura con KaTeX sin dañar el HTML
                        if (el.innerHTML.includes('$')) {
                            const raw = el.innerHTML;
                            const formatted = raw.replace(/\$([^\$]+)\$/g, (match, formula) => {
                                try {
                                    return katex.renderToString(formula, { throwOnError: false });
                                } catch(e) {
                                    return match;
                                }
                            });
                            el.innerHTML = formatted;
                        }
                    });
                }
            }, 50);
        }
